//Status code check
pm.test("Status is 400 Bad Request", () => {
    console.log("Status Code:", pm.response.code);
    pm.response.to.have.status(400);
});

// Parse JSON response
let res = pm.response.json();
console.log("Full Response Body:", res);

//Error message check
pm.test("Error message is 'Empty request body'", () => {
    pm.expect(res.error, "Error field should exist").to.exist;
    console.log("Error message:", res.error);
    pm.expect(res.error).to.equal("Empty request body");
});
